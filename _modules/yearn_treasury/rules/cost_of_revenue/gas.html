@gas("Strategist Gas")
def is_strategist_gas(tx: TreasuryTx) -> bool:
    if tx.symbol == "ETH":
        if tx.from_nickname == "Disperse.app":
            return tx.hash in _STRATEGIST_GAS_HASHES

        # Returned gas
        if tx.hash in _RETURNED_GAS_HASHES:
            tx.amount *= -1
            tx.value_usd *= -1  # type: ignore [operator]
            commit()
            return True

    return False
