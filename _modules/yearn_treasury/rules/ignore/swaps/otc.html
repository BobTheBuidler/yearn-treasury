@swaps("OTC")
def is_otc_swap(tx: TreasuryTx) -> bool:
    txhash = tx.hash
    log_index = tx.log_index
    for otc_swap in _OTC_SWAPS:
        if isinstance(otc_swap, tuple):
            if txhash == otc_swap[0] and log_index in otc_swap[1]:
                return True
        else:
            if txhash == otc_swap:
                return True
    return False
